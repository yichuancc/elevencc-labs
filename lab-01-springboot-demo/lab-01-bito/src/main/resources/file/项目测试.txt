3、sql语句
    性能检查：Performance Check
    or与union all，将or条件转换为union all，利用多个or条件上的索引以提高性能

    SELECT
    	count( id ) AS num_count
    FROM
    	tb_push_message
    WHERE
    	uid = 8212123
    	AND ( type1 = 1 or type2 = 2 or type3 = 3 ) and read_yn = 0


    SELECT SUM( id )
    FROM
    (
        SELECT
            COUNT( id )
        FROM
            tb_push_message
        WHERE
            uid = 8212123 AND read_yn = 0 AND TYPE1 = 1
        UNION ALL
        SELECT
            COUNT( id )
        FROM
            tb_push_message
        WHERE
            uid = 8212123 AND read_yn = 0  AND type2 = 2
        UNION ALL
        SELECT
            COUNT( id )
        FROM
            tb_push_message
        WHERE
            uid = 8212123 AND read_yn = 0 AND TYPE3 = 3
    ) t
